<script>
const productos = [

    {nombre:"Holanda", precio:7500, img:"holanda.jpg"},

    {nombre:"Criollos medianos", precio:4250, img:"criollos.jpg"},
    {nombre:"Criollitos", precio:3800, img:"criollitos.jpg"},
    {nombre:"Queso criollo en barra", precio:8300, img:"barra.jpg"},
    {nombre:"Provoleta", precio:4000, img:"provoleta.jpg"},
    {nombre:"Criollitos al vacío", precio:4100, img:"vacio.jpg"},
    {nombre:"Quesillos x10", precio:16000, img:"quesillos10.jpg"},
    {nombre:"Higos en almíbar", precio:3300, img:"higos.jpg"},
    {nombre:"Criollo grande", precio:7300, img:"grande.jpg"},
    {nombre:"Rollitos de quesillo", precio:1800, img:"rollitos.jpg"},
    {nombre:"Chacarero bastón", precio:6000, img:"chacarero.jpg"},
    {nombre:"Salamines picado fino", precio:3600, img:"salamin.jpg"},
    {nombre:"Rollitos x5", precio:8500, img:"rollitos5.jpg"},
    {nombre:"Queso criollo", precio:7000, img:"criollo.jpg"},
    {nombre:"Salame criollo bastón", precio:4500, img:"salame.jpg"}
];

let carrito = Array(productos.length).fill(0);

function render() {
    const container = document.getElementById("products");
    container.innerHTML = "";

    productos.forEach((p, i) => {
        container.innerHTML += `
        <div class="product">
            <div class="left">
                <img src="${p.img}" class="foto" onerror="this.style.display='none'">
                <div class="info-prod">
                    <div class="name">${p.nombre}</div>
                    <div class="price">$${p.precio}</div>
                </div>
            </div>
            <div class="controls">
                <button class="btn-qty" onclick="cambiar(${i},-1)">-</button>
                <div class="cantidad">${carrito[i]}</div>
                <button class="btn-qty" onclick="cambiar(${i},1)">+</button>
                <button class="btn-qty" style="margin-left:5px; background:#444" onclick="cambiar(${i},10)">+10</button>
            </div>
        </div>`;
    });
}

function cambiar(i, q) {
    carrito[i] += q;
    if (carrito[i] < 0) carrito[i] = 0;
    actualizarTotal();
    render();
}

function actualizarTotal() {
    let t = 0;
    productos.forEach((p, i) => t += p.precio * carrito[i]);
    document.getElementById("total").innerText = t.toLocaleString('es-AR');
}

function enviarPedido() {
    if (!carrito.some(c => c > 0)) return alert("Agrega al menos un producto");
    
    let nombre = document.getElementById("nombre").value || "Cliente";
    let m = `🧀 *PEDIDO - LA FÁBRICA*\n👤 Cliente: ${nombre}\n--------------------------\n`;
    
    productos.forEach((p, i) => {
        if (carrito[i] > 0) m += `• ${p.nombre} x${carrito[i]} ($${(p.precio*carrito[i]).toLocaleString('es-AR')})\n`;
    });

    m += `\n💰 *Total: $${document.getElementById("total").innerText}*`;
    let d = document.getElementById("detalle").value;
    if (d) m += `\n\n📝 Nota: ${d}`;

    window.open("https://wa.me/5493816062471?text=" + encodeURIComponent(m));
}

function imprimirPedido() {
    if (!carrito.some(c => c > 0)) return alert("No hay productos seleccionados");
    
    let nombre = document.getElementById("nombre").value || "CLIENTE";
    let fecha = new Date().toLocaleString('es-AR');
    let total = document.getElementById("total").innerText;
    
    let ticketHtml = `
        <div style="text-align:center; font-family:monospace; color:black;">
            <h2>LA FÁBRICA</h2>
            <p>Artesanales de Calidad</p>
            <p>----------------------------</p>
            <p>FECHA: ${fecha}</p>
            <p>CLIENTE: ${nombre}</p>
            <p>----------------------------</p>
            <table style="width:100%">
    `;

    productos.forEach((p, i) => {
        if (carrito[i] > 0) {
            ticketHtml += `
                <tr>
                    <td style="text-align:left">${p.nombre} x${carrito[i]}</td>
                    <td style="text-align:right">$${(p.precio*carrito[i]).toLocaleString('es-AR')}</td>
                </tr>`;
        }
    });

    ticketHtml += `
            </table>
            <p>----------------------------</p>
            <h3 style="display:flex; justify-content:space-between"><span>TOTAL</span> <span>$${total}</span></h3>
            <p>----------------------------</p>
            <p>¡Muchas gracias por su compra!</p>
        </div>
    `;

    document.getElementById("print-container").innerHTML = ticketHtml;
    window.print();
}

render();
