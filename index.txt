<script>
const productos = [

    {nombre: "Salamines picado fino", precio: 3600, img: "salamin.jpg"},
    {nombre: "Rollitos x5", precio: 8500, img: "rollitos5.jpg"},
    {nombre: "Queso criollo", precio: 7000, img: "criollo.jpg"},
    {nombre: "Salame criollo bastón", precio: 4500, img: "salame.jpg"}
];

let carrito = Array(productos.length).fill(0);

function render() {
    const container = document.getElementById("products");
    container.innerHTML = "";

    productos.forEach((p, i) => {
        container.innerHTML += `
        <div class="product">
            <div class="left">
                <img src="${p.img}" class="foto">
                <div>
                    <div class="name">${p.nombre}</div>
                    <div class="price">$${p.precio}</div>
                </div>
            </div>
            <div class="controls">
                <button onclick="cambiar(${i}, -10)">-10</button>
                <button onclick="cambiar(${i}, -1)">-</button>

                <div class="cantidad">${carrito[i]}</div>

                <button onclick="cambiar(${i}, 1)">+</button>
                <button onclick="cambiar(${i}, 10)">+10</button>
            </div>
        </div>
        `;
    });
}

function cambiar(i, cantidad) {
    carrito[i] += cantidad;
    if (carrito[i] < 0) carrito[i] = 0;
    actualizarTotal();
    render();
}

function actualizarTotal() {
    let total = 0;
    productos.forEach((p, i) => {
        total += p.precio * carrito[i];
    });
    document.getElementById("total").innerText = total;
}

function generarTicket() {
    let hayProductos = carrito.some(c => c > 0);
    if (!hayProductos) {
        alert("No agregaste ningún producto.");
        return null;
    }

    let nombre = document.getElementById("nombre").value || "CLIENTE";
    let detalle = document.getElementById("detalle").value;
    let total = document.getElementById("total").innerText;
    let fecha = new Date().toLocaleString('es-AR');

    let html = `
    <div class="print-ticket">
        <h2>LA FÁBRICA</h2>
        <div class="separator"></div>
        <div style="text-align: center; margin: 2mm 0;">
            <strong>${nombre}</strong><br>
            ${fecha}
        </div>
        <div class="separator"></div>
    `;

    productos.forEach((p, i) => {
        if (carrito[i] > 0) {
            let subtotal = p.precio * carrito[i];
            html += `
            <div class="item">
                <span>${p.nombre}</span>
                <span>x${carrito[i]}</span>
            </div>
            <div class="item" style="text-align: right; font-size: 8pt; margin-bottom: 2mm;">
                $${p.precio} c/u = $${subtotal}
            </div>
            `;
        }
    });

    html += `
        <div class="separator"></div>
        <div class="total-line">
            <span>TOTAL:</span>
            <span>$${total}</span>
        </div>
    `;

    if (detalle) {
        html += `
        <div style="margin-top: 3mm; font-size: 9pt; border-top: 1px dashed black; padding-top: 2mm;">
            <strong>Aclaraciones:</strong><br>
            ${detalle}
        </div>
        `;
    }

    html += `
        <div class="footer">
            <br><br>
            Gracias por su compra
            <br>¡Que disfrutes!
        </div>
    </div>
    `;

    return html;
}

function imprimirPedido() {
    let ticket = generarTicket();
    if (!ticket) return;

    let printContainer = document.getElementById("print-container");
    printContainer.innerHTML = ticket;

    setTimeout(() => {
        window.print();
    }, 100);
}

function enviarPedido() {
    let hayProductos = carrito.some(c => c > 0);
    if (!hayProductos) {
        alert("No agregaste ningún producto.");
        return;
    }

    let nombre = document.getElementById("nombre").value;
    let detalle = document.getElementById("detalle").value;

    let mensaje = "🧀 Pedido - La Fábrica:\n\n";

    if (nombre) mensaje += "👤 " + nombre + "\n\n";

    productos.forEach((p, i) => {
        if (carrito[i] > 0) {
            mensaje += `${p.nombre} x${carrito[i]}\n`;
        }
    });

    mensaje += `\n💰 Total: $${document.getElementById("total").innerText}`;

    if (detalle) mensaje += "\n\n📝 " + detalle;

    const telefono = "5493816062471";
    window.open("https://wa.me/" + telefono + "?text=" + encodeURIComponent(mensaje));
}

render();
